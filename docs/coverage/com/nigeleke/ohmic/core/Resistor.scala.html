42 object Resistor:
43   def setFor(series: Seq[Base]) = (for
44     base       <- series
45     multiplier <- Multiplier.values
46   yield Resistor(base, multiplier)).toSet
47 
48   lazy val fullSet =
49     val emptySet           = Set.empty[Resistor]
50     val unionSeriesWithSet = (set: Set[Resistor], series: Seq[Base]) => set ++ setFor(series)
51     Series.all.foldLeft(emptySet)(unionSeriesWithSet)
52 
53   extension (r: Resistor)
54     def value: ElectricalResistance = Ohms(r.multiplier(r.base))
55 
56     def formattedValue: ElectricalResistance =
57       val unitGroup = summon[UnitGroup[ElectricalResistance]]
58       val formatter = DefaultFormatter(unitGroup)
59       formatter.inBestUnit(value)
60 
61     def colourCodes: Seq[BandColour] =
62       val baseValueString  = r.base.toString
63       val bandColours      = baseValueString.map(BandColour.from)
64       val multiplierColour = r.multiplier.bandColour
65       bandColours :+ multiplierColour
